! ==============================================================================
! PORTFOLIO LAB: ENTERPRISE LAN INFRASTRUCTURE
! DEVICE: ACCESS SWITCH 1 (ASW1)
! ROLE: Layer 2 Access Layer - Host Connectivity & Edge Security
! ==============================================================================
!
version 15.0
no service timestamps log datetime msec
no service timestamps debug datetime msec
no service password-encryption
!
hostname ASW1
!
! [SECURITY HARDENING]
! Removed raw MD5/Type 5 enable secret hash for repository security
enable secret <SANITIZED_REMOVED_FOR_PORTFOLIO>
no ip domain-lookup
!
! [SPANNING TREE PROTOCOL]
spanning-tree mode pvst
spanning-tree extend system-id
!
! [ACTIVE ACCESS INTERFACES: VLAN 10]
! Configured for Engineering hosts with STP Toolkit optimization
interface range FastEthernet0/1 - 2
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
 spanning-tree bpduguard enable
!
! [ACTIVE ACCESS INTERFACES: VLAN 20]
! Configured for Management/IT hosts with STP Toolkit optimization
interface range FastEthernet0/3 - 4
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
 spanning-tree bpduguard enable
!

! [SWITCH PORT HARDENING: UNUSED INTERFACES]
! Rationale: Moving all unused ports to a dead/blackhole VLAN (VLAN 1001) 
! and shutting them down drastically reduces the switch's attack surface.
interface range FastEthernet0/5 - 24
 description ## NOT IN USE - BLACKHOLED ##
 switchport mode access
 switchport access vlan 1001
 shutdown
!
! [UPLINK TRUNK TO DISTRIBUTION SWITCH (DSW1)]
! Carries active data VLANs while isolating the untagged native traffic to VLAN 1001
interface GigabitEthernet0/1
 switchport mode trunk
 switchport trunk native vlan 1001
 switchport trunk allowed vlan 10,20
 spanning-tree portfast disable
!
! [UNUSED UPLINK INTERFACE]
interface GigabitEthernet0/2
 spanning-tree portfast disable
 shutdown
!
! [DEFAULT MANAGEMENT INTERFACE]
interface Vlan1
 no ip address
 shutdown
!
! [LINE CONFIGURATION]
line con 0
!
line vty 0 4
 login
line vty 5 15
 login
!
end
